		name = HashMPScandinavia_5_a

		add_prestige = 10
		add_treasury = 100

			target = NOR
			type = cb_restore_personal_union
		}

		skaneland_area = {
			add_core = ROOT
		}

		subarctic_islands_area = {
			add_permanent_claim = ROOT
		}
	}

		name = HashMPScandinavia_5_b

		add_manpower = 3
		add_sailors = 500

		subarctic_islands_area = {
			add_permanent_claim = ROOT
		}
	}
}

##Succession of Denmark
country_event = {
	id = HashMPScandinavia.6
	title = HashMPScandinavia_6_title
	desc = HashMPScandinavia_6_desc
	picture = DEATH_OF_HEIR_eventPicture
	
	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = HashMPScandinavia_6_a
		hidden_effect = {
			if = {
				limit = {
					has_consort_flag = h_dorothea_consort
				}
				exile_consort_as = h_dorothea
			}
		}
		define_ruler = {
			name = "Christian I"
			dynasty = "von Oldenburg"
			religion = ROOT
			culture = hannoverian
			adm = 4
			dip = 4
			mil = 4
			age = 33
			claim = 80
		}
		if = {
			limit = {
				has_consort_flag = h_dorothea_consort
			}
			set_consort = h_dorothea
		}
	}

	option = {
		name = HashMPScandinavia_6_b
		trigger = {
			has_consort_flag = h_dorothea_consort
		}
		define_ruler = {
			name = "Dorothea I"
			dynasty = "von Hohenzollern"
			religion = ROOT
			culture = saxon
			adm = 5
			dip = 5
			mil = 2
			claim = 60
			age = 19
			female = yes
		}
	}
}
