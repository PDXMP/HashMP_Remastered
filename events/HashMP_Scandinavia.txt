##Aftermath of the Kalmar War
country_event = {
	id = HashMP_Scandinavia.5
	title = HashMP_Scandinavia.5.title
	desc = HashMP_Scandinavia.5.desc
	picture = CIVIL_WAR_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = HashMP_Scandinavia.5.a
		##Vow to restore the union
		add_prestige = 10
		add_treasury = 100

		add_casus_belli = {
			target = NOR
			type = cb_restore_personal_union
		}

		skaneland_area = {
			add_core = ROOT
		}

		25 = {
			add_permanent_claim = ROOT
		}

		subarctic_islands_area = {
			add_permanent_claim = ROOT
		}

		custom_tooltip = h_danes_kalmar_event_tooltip

		hidden_effect = {
			set_country_flag = h_restore_the_union
			set_country_flag = h_kalmar_union_event_choice_concluded
		}
	}
	
	option = {
		name = HashMP_Scandinavia.5.b
		##Forget the union

		add_prestige = -5
		add_stability_or_adm_power = yes
		add_manpower = 3
		add_sailors = 500

		subarctic_islands_area = {
			add_permanent_claim = ROOT
		}

		hidden_effect = {
			set_country_flag = h_forget_the_union
			set_country_flag = h_kalmar_union_event_choice_concluded
		}
	}
}
